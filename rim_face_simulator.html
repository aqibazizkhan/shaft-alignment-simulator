<script>

function val(id){ return parseFloat(document.getElementById(id).value)||0; }

function calculate(){

    // RIM

    let RT=val("rt"), RB=val("rb"), RL=val("rl"), RR=val("rr");

    // FACE

    let FT=val("ft"), FB=val("fb"), FL=val("fl"), FR=val("fr");

    // Geometry

    let dia=val("dia");

    let A=val("A");

    let B=val("B");

    let C=val("C");

    // Vertical plane

    let rimV = (RT - RB)/2;

    let faceV = (FT - FB)/dia;

    let offsetV = rimV;

    let angleV = faceV;

    // Horizontal plane

    let rimH = (RL - RR)/2;

    let faceH = (FL - FR)/dia;

    let offsetH = rimH;

    let angleH = faceH;

    // Shim calculations (vertical only)

    let shimFront = offsetV + angleV * B;

    let shimRear  = offsetV + angleV * C;

    // Output

    document.getElementById("voffset").innerText = "Vertical Offset: " + offsetV.toFixed(3);

    document.getElementById("vangle").innerText  = "Vertical Angularity: " + angleV.toFixed(5);

    document.getElementById("hoffset").innerText = "Horizontal Offset: " + offsetH.toFixed(3);

    document.getElementById("hangle").innerText  = "Horizontal Angularity: " + angleH.toFixed(5);

    document.getElementById("frontShim").innerText = "Front Foot Correction: " + shimFront.toFixed(3) + " mm";

    document.getElementById("rearShim").innerText  = "Rear Foot Correction: " + shimRear.toFixed(3) + " mm";

    // Tolerance check

    let tol = 0.05;

    if(Math.abs(offsetV)<tol && Math.abs(angleV)<0.0005)

        document.getElementById("status").innerText = "Aligned";

    else

        document.getElementById("status").innerText = "Misaligned";

    draw(offsetV, angleV);

}

function draw(offset, angle){

    let c=document.getElementById("canvas");

    let ctx=c.getContext("2d");

    ctx.clearRect(0,0,700,300);

    // Fixed machine

    ctx.beginPath();

    ctx.moveTo(50,150);

    ctx.lineTo(300,150);

    ctx.strokeStyle="black";

    ctx.stroke();

    // Misaligned machine

    ctx.beginPath();

    ctx.moveTo(300,150+offset*40);

    ctx.lineTo(600,150+offset*40+angle*2000);

    ctx.strokeStyle="red";

    ctx.stroke();

}

function randomCase(){

    document.getElementById("rt").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("rb").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("rl").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("rr").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("ft").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("fb").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("fl").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("fr").value=(Math.random()-0.5).toFixed(3);

    document.getElementById("dia").value=200;

    document.getElementById("A").value=150;

    document.getElementById("B").value=300;

    document.getElementById("C").value=600;

}
